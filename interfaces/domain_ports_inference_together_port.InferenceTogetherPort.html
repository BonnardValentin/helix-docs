Interface InferenceTogetherPort

Interface for Inference Together operations

This port defines the standard interface for collaborative inference that must be implemented by any inference adapter in the application.

interface InferenceTogetherPort {
    getInferenceResults(agentId): Promise<Either<Error, InferenceResult[]>>;
    infer(request): Promise<Either<Error, InferenceResult[]>>;
    initialize(): Promise<Either<Error, void>>;
    registerAgent(agent): Promise<Either<Error, void>>;
    unregisterAgent(agentId): Promise<Either<Error, void>>;
}
