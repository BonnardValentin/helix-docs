Interface AgentRepositoryPort

Interface for agent repository operations

This port defines the standard interface for agent data persistence that must be implemented by any agent repository adapter in the application.

interface AgentRepositoryPort {
    delete(id): Promise<Either<boolean>>;
    findAll(): Promise<Either<Agent[]>>;
    findByCapabilities(capabilities): Promise<Either<Agent[]>>;
    findById(id): Promise<Either<null | Agent>>;
    findByMetadata(metadata): Promise<Either<Agent[]>>;
    findByStatus(status): Promise<Either<Agent[]>>;
    save(agent): Promise<Either<Agent>>;
    update(agent): Promise<Either<Agent>>;
}
